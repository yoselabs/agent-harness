# ai-harness MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A single CLI tool that detects project stacks, runs all harness checks (conftest + hadolint + yamllint + ruff + file length), and initializes new projects with the right configs — replacing hand-maintained Makefiles.

**Architecture:** Python CLI (click) distributed as a uv script with inline dependencies. Policies bundled in the package. Config via `.agent-harness.yml` in project root. Stack detection by file presence. Each check runs as a subprocess, output unified. Must complete `lint` in <1s on a typical project.

**Tech Stack:** Python 3.12+, click, PyYAML, uv (for running/distributing). Conftest, hadolint, yamllint, ruff as external tools (not embedded).

**Test project:** ~/Workspaces/aggre (Python + Docker, already has harness applied manually).

```python
# src/ai_harness/detect.py
from pathlib import Path

STACK_INDICATORS = {
    "python": ["pyproject.toml", "setup.py", "requirements.txt", "uv.lock"],
    "docker": ["Dockerfile", "docker-compose.prod.yml", "docker-compose.yml"],
}

def detect_stacks(project_dir: Path) -> set[str]:
    """Detect which stacks a project uses based on file presence."""
    stacks = set()
    for stack, indicators in STACK_INDICATORS.items():
        for indicator in indicators:
            if (project_dir / indicator).exists():
                stacks.add(stack)
                break
    return stacks
```

```python
# src/ai_harness/runner.py
from __future__ import annotations
import subprocess
import shutil
from dataclasses import dataclass

@dataclass
class CheckResult:
    name: str
    passed: bool
    output: str = ""
    error: str = ""
    duration_ms: int = 0

def run_check(name: str, cmd: list[str], cwd: str | None = None) -> CheckResult:
    """Run a check command and return structured result."""
    tool = cmd[0]
    if not shutil.which(tool):
        return CheckResult(name=name, passed=False, error=f"{tool} not found — not installed or not in PATH")

    import time
    start = time.monotonic()
    try:
        result = subprocess.run(cmd, capture_output=True, text=True, cwd=cwd, timeout=60)
        duration = int((time.monotonic() - start) * 1000)
        return CheckResult(
            name=name,
            passed=result.returncode == 0,
            output=result.stdout,
            error=result.stderr,
            duration_ms=duration,
        )
    except subprocess.TimeoutExpired:
        return CheckResult(name=name, passed=False, error=f"{name} timed out after 60s")
```

```python
# src/ai_harness/config.py
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
import yaml
from ai_harness.detect import detect_stacks

@dataclass
class PythonConfig:
    coverage_threshold: int = 95
    line_length: int = 140
    max_file_lines: int = 500

@dataclass
class DockerConfig:
    own_image_prefix: str = ""

@dataclass
class HarnessConfig:
    stacks: set[str] = field(default_factory=set)
    python: PythonConfig = field(default_factory=PythonConfig)
    docker: DockerConfig = field(default_factory=DockerConfig)

def load_config(project_dir: Path) -> HarnessConfig:
    """Load .agent-harness.yml or fall back to auto-detection + defaults."""
    config = HarnessConfig()
    cfg_path = project_dir / ".agent-harness.yml"

    if cfg_path.exists():
        raw = yaml.safe_load(cfg_path.read_text()) or {}
        if "stacks" in raw:
            config.stacks = set(raw["stacks"])
        if "python" in raw:
            for k, v in raw["python"].items():
                if hasattr(config.python, k):
                    setattr(config.python, k, v)
        if "docker" in raw:
            for k, v in raw["docker"].items():
                if hasattr(config.docker, k):
                    setattr(config.docker, k, v)

    if not config.stacks:
        config.stacks = detect_stacks(project_dir)

    return config
```

```python
# src/ai_harness/checks/conftest.py
from __future__ import annotations
from pathlib import Path
from ai_harness.runner import run_check, CheckResult

POLICIES_DIR = Path(__file__).parent.parent.parent.parent / "policies"

def run_conftest_dockerfile(project_dir: Path) -> CheckResult:
    dockerfile = project_dir / "Dockerfile"
    if not dockerfile.exists():
        return CheckResult(name="conftest:dockerfile", passed=True, output="No Dockerfile found, skipping")
    policy_dir = str(POLICIES_DIR / "dockerfile")
    return run_check(
        "conftest:dockerfile",
        ["conftest", "test", str(dockerfile), "--parser", "dockerfile", "-p", policy_dir, "--all-namespaces"],
        cwd=str(project_dir),
    )

def run_conftest_compose(project_dir: Path, own_image_prefix: str = "") -> CheckResult:
    compose = project_dir / "docker-compose.prod.yml"
    if not compose.exists():
        return CheckResult(name="conftest:compose", passed=True, output="No docker-compose.prod.yml found, skipping")
    policy_dir = str(POLICIES_DIR / "compose")
    cmd = ["conftest", "test", str(compose), "-p", policy_dir, "--all-namespaces"]
    if own_image_prefix:
        cmd.extend(["--data", f'{{"own_image_prefix": "{own_image_prefix}"}}'])
    return run_check("conftest:compose", cmd, cwd=str(project_dir))

def run_conftest_python(project_dir: Path) -> CheckResult:
    pyproject = project_dir / "pyproject.toml"
    if not pyproject.exists():
        return CheckResult(name="conftest:python", passed=True, output="No pyproject.toml found, skipping")
    policy_dir = str(POLICIES_DIR / "python")
    return run_check("conftest:python", ["conftest", "test", str(pyproject), "-p", policy_dir, "--all-namespaces"], cwd=str(project_dir))

def run_conftest_gitignore(project_dir: Path) -> CheckResult:
    gitignore = project_dir / ".gitignore"
    if not gitignore.exists():
        return CheckResult(name="conftest:gitignore", passed=True, output="No .gitignore found, skipping")
    policy_dir = str(POLICIES_DIR / "gitignore")
    return run_check("conftest:gitignore", ["conftest", "test", str(gitignore), "--parser", "ignore", "-p", policy_dir, "--all-namespaces"], cwd=str(project_dir))

def run_conftest_json(project_dir: Path) -> CheckResult:
    """Validate JSON files via conftest parse."""
    import subprocess
    result = subprocess.run(["git", "ls-files", "*.json", "**/*.json"], capture_output=True, text=True, cwd=str(project_dir))
    json_files = [f for f in result.stdout.strip().splitlines() if f]
    if not json_files:
        return CheckResult(name="conftest:json", passed=True, output="No JSON files found")
    return run_check("conftest:json", ["conftest", "parse", "--parser", "json"] + json_files, cwd=str(project_dir))
```

```python
# src/ai_harness/lint.py
from pathlib import Path
from ai_harness.config import load_config
from ai_harness.runner import CheckResult
from ai_harness.checks.conftest import (
    run_conftest_dockerfile, run_conftest_compose,
    run_conftest_python, run_conftest_gitignore, run_conftest_json,
)
from ai_harness.checks.hadolint import run_hadolint
from ai_harness.checks.yamllint_check import run_yamllint
from ai_harness.checks.ruff import run_ruff
from ai_harness.checks.file_length import run_file_length

def run_lint(project_dir: Path) -> list[CheckResult]:
    config = load_config(project_dir)
    results: list[CheckResult] = []

    # Universal checks
    results.append(run_conftest_gitignore(project_dir))
    results.append(run_conftest_json(project_dir))
    results.append(run_yamllint(project_dir))

    # Python checks
    if "python" in config.stacks:
        results.append(run_ruff(project_dir))
        results.append(run_conftest_python(project_dir))
        results.append(run_file_length(project_dir, config.python.max_file_lines))

    # Docker checks
    if "docker" in config.stacks:
        results.append(run_conftest_dockerfile(project_dir))
        results.append(run_conftest_compose(project_dir, config.docker.own_image_prefix))
        results.append(run_hadolint(project_dir))

    return results
```

```python
# In cli.py lint command:
@cli.command()
def lint():
    """Run all harness checks."""
    results = run_lint(Path.cwd())

    failed = [r for r in results if not r.passed]
    passed = [r for r in results if r.passed]
    total_ms = sum(r.duration_ms for r in results)

    for r in results:
        icon = "PASS" if r.passed else "FAIL"
        click.echo(f"  {icon}  {r.name} ({r.duration_ms}ms)")
        if not r.passed and r.error:
            for line in r.error.strip().splitlines():
                click.echo(f"       {line}")

    click.echo(f"\n{len(passed)} passed, {len(failed)} failed ({total_ms}ms)")
    raise SystemExit(1 if failed else 0)
```
